        .text
        .globl main

main:   li      $v0, 5
        syscall

        move    $s0, $v0

        la      $a0, array
        la      $a1, array
        li      $s1, 0
FOR1:   beq     $s0, $s1, END
        li      $v0, 5
        syscall
        sw      $v0, 0($a1)
        addi    $s1, 1
        addiu   $a1, 4
        j       FOR

END1:
        addiu   $s0, $a1, -8    # q = end - 2
        
FOR:    bltu    $s0, $a0, END   
        move    $s1, $a0
        
for:    bgtu    $s1, $s0, end
        lw      $t0, 4($s1)
        lw      $t1, 0($s1)
        
        bgeu    $t1, $t0, continue
        sw      $t1, 4($s1)
        sw      $t0, 0($s1)
        
continue:       addiu   $s1, 4
        j       for
end:    addiu   $s0, $s0, -1
        j       FOR
END:
for1:   beq     $a0, $a1, end1
        lw      $a0, 0($a0)
        li      $v0, 1
        syscall

        la      $a0, space
        li      $v0, 4
        syscall

        addiu   $a0, $a0, 1
        j       for1

end1:   li      $v0, 10
        syscall

        .data
